# You can use this function for plotting raster plots (spikes), or you can use your own defined function
def raster(event_times_list, color='k', t_max=10.):
    """
    Creates a raster plot
    Parameters
    ----------
    event_times_list : iterable
                       a list of event time iterables
    color : string
            color of vlines
    Returns
    -------
    ax : an axis containing the raster plot
    https://scimusing.wordpress.com/2013/05/06/making-raster-plots-in-python-with-matplotlib/
    """
    plt.figure(figsize=(10,3));
    ax = plt.gca();
    for ith, trial in enumerate(event_times_list):
        plt.vlines(trial, ith + .5, ith + 1.5, color=color);
    plt.ylim(.5, len(event_times_list) + .5);
    ax.set_xlim((0,1));
    return ax

@jit(nopython = True)
def spike_gen_poiss(t, lam_max):
    
    spike_train = np.zeros(1000);
    spike_times = np.zeros(100);
    iter = 0;
    
    for i in range(len(t)-1):         
        mean = lam_max * ((t[i+1]**2)/2 - (t[i+1]**3)/3 - (t[i]**2)/2 + (t[i]**3)/3);
        p = mean * np.exp(-mean);
        s = np.random.uniform(0,1,1);
        
        if s[0] < p:
            spike_train[i] = t[i];
            
        if (iter + 1) * 0.01 > t[i]:
            if s[0] < p:
                spike_times[iter] += 1;
        else:
            iter += 1;
            if s[0] < p:
                spike_times[iter] += 1;
    return spike_train, spike_times

@jit(nopython = True)
def spike_count_dist(t2, t1):
    mean_bin = lam_max * ((t2**2)/2 - (t2**3)/3 - (t1**2)/2 + (t1**3)/3);
    k_spikes = np.arange(0, np.ceil(mean_bin)*2+1);
    spike_count_dist_bin = np.zeros(len(k_spikes));
        
    for l in range(len(k_spikes)):
        fact = 1;
        if k_spikes[l] != 0:
            fact = np.prod(k_spikes[1:l+1]);
            
        p = np.power(mean_bin, k_spikes[l]) * np.exp(-mean_bin) / fact;
        spike_count_dist_bin[l] = p;

    spike_count_dist_bin[len(k_spikes)-1] = spike_count_dist_bin[len(k_spikes)-1] + (1 - np.sum(spike_count_dist_bin));
    
    return k_spikes, spike_count_dist_bin, mean_bin

@jit(nopython = True)
def time_stamp_uniform_dist(t2, t1, mean_bin, m):
    
    spike_t_bin = np.arange(t1, t2, T/(m*100));
    
    spike_times_from_density_bin = np.zeros(len(spike_t_bin));

    for l in range(len(spike_t_bin)-1):
        lam = lam_max * (spike_t_bin[l] - spike_t_bin[l]**2);
        p = lam/mean_bin;
        spike_times_from_density_bin[l] = p/(m*100);
        
    if np.sum(spike_times_from_density_bin) > 1:
        sub = (np.sum(spike_times_from_density_bin) - 1);
        
        i = len(spike_times_from_density_bin) - 1;
        while i >= 0:
            if spike_times_from_density_bin[i] > sub:
                spike_times_from_density_bin[i] = spike_times_from_density_bin[i] - sub;      
                break;
            i = i -1;
    else:
        spike_times_from_density_bin[len(spike_t_bin)-1] =1-np.sum(spike_times_from_density_bin);
    
    return spike_t_bin, spike_times_from_density_bin

c) Thinning method: First generate a spike train from a Poisson process of rate $\lambda_{max}$ and then iterate through all spikes $l = 1, \ldots, k$ and randomly delete them with probability $1 - \lambda(t^\star_l) /\lambda_{max}$, where $t^\star_l$ is a time point of the $l$-th spike (before thinning). (2 points)

Homogeneous Poisson spike generation function, for each interval if uniform random variable is less than the probability of generating one spike, I generated a spike. I have used following reference: http://www.cns.nyu.edu/~david/handouts/poisson.pdf

@jit(nopython = True)
def spike_gen_poiss_homo(t, lam_max, dt):
    
    spike_train = np.zeros(len(t));
    
    iter = 0;        
    mean = lam_max * dt;
    
    for i in range(len(t)-1):     
        p = mean * np.exp(-mean);
        s = np.random.uniform(0,1,1);
        
        if s[0] < p:
            spike_train[i] = t[i];
            
    return spike_train
